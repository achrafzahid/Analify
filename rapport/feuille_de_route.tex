\textbf{Frontend} & 
\textbf{React 18} avec \textbf{TypeScript}
\begin{itemize}[leftmargin=*, noitemsep, topsep=0pt]
    \item \textbf{Vite} : Build tool et bundler
    \item \textbf{Tailwind CSS} : Framework CSS utility-first
    \item \textbf{shadcn/ui} : Composants UI modernes
    \item \textbf{Bun} : Package manager rapide
    \item \textbf{Axios} : Communication HTTP
    \item \textbf{React Router} : Navigation SPA
\end{itemize}
\\
\hline

\textbf{Backend} & 
\textbf{Java 21} avec \textbf{Spring Boot 3.4.13}
\begin{itemize}[leftmargin=*, noitemsep, topsep=0pt]
    \item \textbf{Spring Data JPA} : ORM et repositories
    \item \textbf{Spring Security} : Sécurité et authentification
    \item \textbf{Spring AI} : Intégration LLM (Ollama)
    \item \textbf{Spring Validation} : Validation (@Valid)
    \item \textbf{Lombok} : Réduction du code boilerplate
    \item \textbf{MapStruct} : Mapping DTO ↔ Entités
\end{itemize}
\\
\hline

\textbf{Architecture} & 
\textbf{4 couches} : Frontend (React) → Controller → Service → Repository
\\
\hline

\textbf{Base de données} & 
\textbf{PostgreSQL 16} : 18 tables normalisées, 1M+ lignes, indexation optimisée
\\
\hline

\textbf{Sécurité} & 
\textbf{JWT} : Authentification stateless, expiration 24h, BCrypt pour mots de passe
\\
\hline

\textbf{Rôles utilisateurs} & 
\textbf{ADMIN\_G} (global), \textbf{ADMIN\_STORE} (magasin), \textbf{INVESTOR} (bidding), \textbf{CAISSIER} (caisse)
\\
\hline

\textbf{Intelligence Artificielle} & 
\textbf{Ollama - LLaMA 3.2:3b}
\begin{itemize}[leftmargin=*, noitemsep, topsep=0pt]
    \item Contexte : 8192 tokens
    \item GPU : NVIDIA RTX 4060 (8GB VRAM)
    \item Assistant conversationnel en langage naturel
    \item Historique : 5 derniers échanges
\end{itemize}
\\
\hline

\textbf{Conteneurisation} & 
\textbf{Docker \& Docker Compose}
\begin{itemize}[leftmargin=*, noitemsep, topsep=0pt]
    \item Multi-stage builds optimisés
    \item 4 services : backend, frontend, postgres, ollama
    \item Health checks et volumes persistants
\end{itemize}
\\
\hline

\textbf{Build \& Déploiement} & 
\textbf{Maven} (backend), \textbf{Nginx} (frontend production), \textbf{Git} (versioning)
\\
\hline

\textbf{Tests} & 
\textbf{Postman} : Tests API REST, validation endpoints et sécurité
\\
\hline

\textbf{Traitement données} & 
\textbf{Python3} avec Jupyter
\begin{itemize}[leftmargin=*, noitemsep, topsep=0pt]
    \item \textbf{Pandas} : Transformation CSV
    \item \textbf{Faker} : Génération données synthétiques
    \item \textbf{SQLAlchemy} : Injection PostgreSQL
\end{itemize}
\\
\hline
